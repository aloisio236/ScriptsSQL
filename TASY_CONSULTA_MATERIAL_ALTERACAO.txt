/* TASY_CONSULTA_MATERIAL_ALTERACAO */

 

SELECT  A.CD_ESTABELECIMENTO CD_ESTAB,
        OBTER_NOME_ESTAB(A.CD_ESTABELECIMENTO) NM_ESTABELECIMENTO,
        A.CD_MATERIAL,
        OBTER_DADOS_MATERIAL(A.CD_MATERIAL,'D') DS_MATERIAL,
        A.NM_TABELA,
        A.NM_ATRIBUTO,
        A.DS_VALOR_ANT,
        A.DS_VALOR_NOVO,
        OBTER_NOME_FUNCAO(A.CD_FUNCAO) NM_FUNCAO,
        OBTER_NOME_PERFIL(A.CD_PERFIL) NM_PERFIL,
        A.DT_ATUALIZACAO,
        A.NM_USUARIO
FROM    MATERIAL_LOG_ALTERACAO A
WHERE A.CD_MATERIAL = :COD_MATERIAL
ORDER BY A.DT_ATUALIZACAO DESC
