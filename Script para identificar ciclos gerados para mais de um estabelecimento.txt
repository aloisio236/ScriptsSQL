SELECT   OBTER_NOME_ESTAB(B.CD_ESTABELECIMENTO) "UNIDADE",
         A.CD_PESSOA_FISICA "PRONTUARIO",
         INITCAP(OBTER_NOME_PF(A.CD_PESSOA_FISICA)) "NOME DO PACIENTE",
--         SEM_ACENTO(OBTER_NOME_PF(A.CD_MEDICO_RESP)) "MEDICO DO PACIENTE",
         ELIMINA_ACENTUACAO(SUBSTR(obter_desc_protocolo_onco(A.NR_SEQ_PACIENTE),1,50)) "NOME DO PROTOCOLO",
--        B.NR_SEQ_ATENDIMENTO,
--        B.NR_SEQ_PACIENTE,
        B.NR_CICLO,
        B.DS_DIA_CICLO,
        B.NR_ATENDIMENTO,
        B.NR_PRESCRICAO,
        B.DT_PREVISTA,
        B.DT_REAL
FROM       PACIENTE_SETOR A
INNER JOIN PACIENTE_ATENDIMENTO B ON A.NR_SEQ_PACIENTE = B.NR_SEQ_PACIENTE
WHERE   A.NR_SEQ_PACIENTE = 12049
ORDER BY B.DT_PREVISTA ASC
